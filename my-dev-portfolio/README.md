# 1. Inicializace nové React aplikace s Vite
npm create vite@latest my-dev-portfolio -- --template react-ts

# 2. Přechod do složky projektu
cd my-dev-portfolio

# 3. Instalace závislostí
npm install


# Run dev 
npm run dev


# Tailwind
npm install -D tailwindcss postcss autoprefixer
npx tailwindcss init -p


# GithubPages
npm install gh-pages --save-dev
npm run deploy
